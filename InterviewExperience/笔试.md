~~~C++
#include <iostream>
#include <vector>
using namespace std;

typedef long long ll;
const int MOD = 1e9 + 7;
typedef vector<vector<ll>> mat;     

mat mul(mat &a, mat &b)
{
    mat ret(a.size(), vector<ll>(b[0].size(), 0));
    // 矩阵乘法
    for(int i = 0; i < a.size(); ++ i)
        for(int j = 0; j < b[0].size(); ++ j)
            for(int k = 0; k < a[0].size(); ++ k)
                ret[i][j] = (ret[i][j] + a[i][k] * b[k][j]) % MOD;

    return ret;
}

mat pow(mat a, ll n)
{
    // 定义并初始化成单位矩阵
    mat res(a.size(), vector<ll>(a[0].size()));
    for(int i = 0; i < a.size(); ++ i)
            res[i][i] = 1;
    
    // 快速幂
    while(n)
    {
        if(n & 1)
            res = mul(res, a);
        a = mul(a, a);
        n /= 2;
    }

    return res;
}

void solve(ll n)
{
    vector<ll> base{3,2,1}; // 最初的元素f(3),f(2),f(1)
    mat mm(3, vector<ll>(3)); // 根据dp公式，构建矩阵
    mm[0][0] = 1, mm[0][1] = 0, mm[0][2] = 1;
    mm[1][0] = 1, mm[1][1] = 0, mm[1][2] = 0;
    mm[2][0] = 0, mm[2][1] = 1, mm[2][2] = 0;
	
    // 矩阵快速幂
    mat pmat = pow(mm, n - 3);
	
    // 求f(n)
    ll ans = 0;
    for(int i = 0; i < 3; ++ i)
        ans = (ans + base[i] * pmat[0][i]) % MOD;
    
    cout << ans;
}

int main() {
    
    long long n;
    cin >> n;
    if(n <= 3)
        cout << n;
    else
        solve(n);
    
    return 0;
}
// 64 位输出请用 printf("%lld")
~~~

~~~C++
#include <queue>
#include <vector>
#include <string>
#include <ctime>
#include <thread>
#include <mutex>
#include <set>
#include <cmath>
#include <iostream>
using namespace std;

struct sortNode // 排序节点
{
    unsigned int score;
    time_t t;
    int rank;
    
    sortNode(unsigned int s = 0) : score(s)
    {
        t = time(NULL);
    }
};
class rankingList // 排行榜
{
private:
    int len = 0; // 上榜人数
    unsigned int minScore = 0; // 前10000名的最小分数，减少锁的使用
    vector<sortNode*> arr; // 排序数组
    set<sortNode*> hash; // hash：判断是否在前10000名
    static rankingList* rLPtr; // 单例指针
    static mutex mtx; // 互斥锁
    
    // 单例模式
    rankingList() {};
    rankingList(const rankingList& rl) = delete;
    rankingList& operator=(const rankingList& rl) = delete;
    ~rankingList() // 析构排序数组中的资源
    {
        for(auto* ptr : arr)
            delete ptr;
    }
public:
    static rankingList* GetRankList() // 获取单例指针
    {
        if(rLPtr == nullptr)
        {
            mtx.lock();
            while(nullptr == rLPtr)
               rLPtr = new rankingList();
            mtx.unlock();
        }
        
        return rLPtr;
    }
    void insert(sortNode* s) // 插入至排序数组
    {
        if(s->score <= minScore) // 小于最小值则直接退出
        {
            delete s;
            return ;
        }
        
        // 多线程情况下只能一个线程在插入
        mtx.lock();
        if(len < 10000) // 不满10000
        {
            s->rank = len;
            arr.push_back(s);
            hash.insert(s);
            // 插入法
            int n = len;
            while(n && arr[n]->score > arr[n - 1]->score)
            {
                swap(arr[n]->rank, arr[n - 1]->rank);
                swap(arr[n], arr[n - 1]);
                -- n;
            }
            ++ len;
        }
        else
        {
            // 最后一个需要删除
            delete arr[9999]; 
            hash.erase(arr[9999]);
            
            arr[9999] = s;
            s->rank = 9999;
            int n = 9999;
            while(n && arr[n]->score > arr[n - 1]->score)
            {
                swap(arr[n]->rank, arr[n - 1]->rank);
                swap(arr[n], arr[n - 1]);
                -- n;
            }
            hash.insert(s);
            minScore = arr[9999]->score;
        }
        mtx.unlock();
    }
    void showRank(sortNode* s)
    {
        mtx.lock();
        if(hash.count(s) == 0) // s在hash中代表在前10000名
        {
            cout << "很遗憾、未上榜!" << endl;
        }
        else if(s->rank > 100)
        {
~~~
